策略说明

GARCH 在高频数据上效果更好。策略：

  1. 在日度数据上拟合 ARIMA-GARCH
  2. 提取日度条件方差 σ²ₜ
  3. 聚合为月度均值，作为 VAR 的”波动率”变量

A3. GARCH(1,1) 拟合

spec <- ugarchspec(
  variance.model = list(
    model      = "sGARCH",
    garchOrder = c(1, 1)
  ),
  mean.model = list(
    armaOrder    = c(1, 1),
    include.mean = TRUE
  ),
  distribution.model = "norm"
)

fit_garch <- ugarchfit(spec = spec, data = fx_daily$log_ret)

A5. 提取条件方差 & 聚合为月度

fx_daily$cond_var <- as.numeric(sigma(fit_garch))^2

vol_monthly <- fx_daily %>%
  mutate(ym = floor_date(date, "month")) %>%
  group_by(ym) %>%
  summarise(
    garch_vol = mean(cond_var, na.rm = TRUE),
    .groups   = "drop"
  ) %>%
  rename(date = ym)

ggplot(vol_monthly, aes(x = date, y = garch_vol)) +
  geom_line(color = "darkorange") +
  labs(title = "GARCH(1,1) Monthly Conditional Variance",
       subtitle = "Aggregated from daily estimates",
       x = NULL, y = expression(sigma^2)) +
  theme_minimal()

Part C: 模型比较（可选）

# EGARCH: 捕捉非对称效应
spec_e <- ugarchspec(
  variance.model = list(model = "eGARCH", garchOrder = c(1, 1)),
  mean.model     = list(armaOrder = c(1, 1), include.mean = TRUE),
  distribution.model = "norm"
)
fit_e <- ugarchfit(spec_e, data = fx_daily$log_ret)

# GJR-GARCH
spec_gjr <- ugarchspec(
  variance.model = list(model = "gjrGARCH", garchOrder = c(1, 1)),
  mean.model     = list(armaOrder = c(1, 1), include.mean = TRUE),
  distribution.model = "norm"
)
fit_gjr <- ugarchfit(spec_gjr, data = fx_daily$log_ret)

# 比较
model_comparison <- tibble(
  Model = c("sGARCH(1,1)", "eGARCH(1,1)", "gjrGARCH(1,1)"),
  AIC   = c(infocriteria(fit_garch)[1],
            infocriteria(fit_e)[1],
            infocriteria(fit_gjr)[1]),
  BIC   = c(infocriteria(fit_garch)[2],
            infocriteria(fit_e)[2],
            infocriteria(fit_gjr)[2])
)
print(model_comparison)
